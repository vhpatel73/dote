Technical Specs

Real-time Integration Guide

How to report AI benefits via Webhook API? External applications can securely report outcomes to update the tracker in real-time.

Overview

The AI Value Board provides a REST-compatible webhook endpoint that allows external AI applications to report their realized benefits automatically. This enables real-time tracking of productivity gains and revenue impact without manual entry.

Endpoint Specification

POST {{ webhook_url }}

Content-Type: application/json

Payload Structure

Field Type Requirement Description
webhook_key String Required The secure secret key for the specific initiative.
kpi_value Float Conditional Required if the initiative tracks Productivity Gain.
revenue_impact Float Conditional Required if the initiative tracks New Business.
month String Optional Format: YYYY-MM. Defaults to current month if omitted.

WebhookIntegration.java 
import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

public class WebhookIntegration {
    public static void main(String[] args) throws Exception {
        String url = "{{ webhook_url }}";
        String payload = "{" +
            "\"webhook_key\": \"{% if initiative %}{{ initiative.webhook_key }}{% else %}YOUR_SECRET_KEY{% endif %}\", " +
            "\"kpi_value\": 150.5, " +
            "\"revenue_impact\": 0, " +
            "\"month\": \"2024-03\"" +
        "}";

        HttpClient client = HttpClient.newHttpClient();
        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(url))
            .header("Content-Type", "application/json")
            .POST(HttpRequest.BodyPublishers.ofString(payload))
            .build();

        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());
        System.out.println(response.body());
    }
}
WebhookCallout.cls 
public class WebhookCallout {
    @future(callout=true)
    public static void sendBenefitReport() {
        Http http = new Http();
        HttpRequest request = new HttpRequest();
        request.setEndpoint('{{ webhook_url }}');
        request.setMethod('POST');
        request.setHeader('Content-Type', 'application/json');
        
        String key = '{% if initiative %}{{ initiative.webhook_key }}{% else %}YOUR_SECRET_KEY{% endif %}';
        String payload = '{"webhook_key": "' + key + '", "kpi_value": 150.5, "revenue_impact": 0, "month": "2024-03"}';
        request.setBody(payload);

        HttpResponse response = http.send(request);
        System.debug(response.getBody());
    }
}
WebhookCallout.gs 
uses java.net.URL
uses java.net.HttpURLConnection
uses java.io.OutputStreamWriter
uses java.io.BufferedReader
uses java.io.InputStreamReader

var endpoint = new URL("{{ webhook_url }}")
var key = "{% if initiative %}{{ initiative.webhook_key }}{% else %}YOUR_SECRET_KEY{% endif %}"

var conn = endpoint.openConnection() as HttpURLConnection
conn.RequestMethod = "POST"
conn.setRequestProperty("Content-Type", "application/json")
conn.DoOutput = true

var payload = "{\"webhook_key\": \"${key}\", \"kpi_value\": 150.5, \"revenue_impact\": 0, \"month\": \"2024-03\"}"

var writer = new OutputStreamWriter(conn.OutputStream)
writer.write(payload)
writer.flush()
writer.close()

var reader = new BufferedReader(new InputStreamReader(conn.InputStream))
var response = new StringBuilder()
var line = reader.readLine()
while (line != null) {
  response.append(line)
  line = reader.readLine()
}
reader.close()

print("Status: ${conn.ResponseCode}")
print(response.toString())
